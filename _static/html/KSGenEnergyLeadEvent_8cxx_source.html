148  double KSGenEnergyLeadEvent::Fermi(double E,double mnu, double E0, double Z)
149  {
150  // This subr. computes the Fermi beta energy spectrum shape,
151  // with neutrino mass
152  // E: electron kinetic energy in eV
153  // mnu: neutrino mass
154  double beta,E1,p1,p2,E2,FC,x,Fermiret;
155  E2=KConst::M_el_eV()+E; // electron total energy
156  p2=sqrt(E2*E2-KConst::M_el_eV()*KConst::M_el_eV()); // electron momentum
157  beta=p2/E2;
158  E1=E0-E; // neutrino total energy
159  if(E1>=mnu)
160  p1=sqrt(fabs(E1*E1-mnu*mnu)); // neutrino momentum
161  else
162  p1=0.;
163  x=2.*KConst::Pi()*Z*KConst::Alpha()/beta;
164  FC=x/(1.-exp(-x)); // Coulomb correction factor
165  Fermiret=p2*E2*p1*E1*FC;
166  return Fermiret;
167  }
