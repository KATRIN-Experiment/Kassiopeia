50  double FBBRionization::RadInt2BoundFreeBurgess(int n, int l, double E, int sign) {
51 
52  int lp=l+sign;
53  if(lp<0)
54  return 0.;
55 
56  if(n<1 || l<0 || l>n-1)
57  {
58  puts("Message from function RadInt2BoundFreeBurgess:");
59  puts("n has to be larger than 0!");
60  puts("l has to be larger than -1 and smaller than n !");
61  puts("Program running is stopped !!! ");
62  exit(1);
63  }
64 
65  if(n>nMAX)
66  {
67  puts("Message from function RadInt2BoundFreeBurgess");
68  puts("n cannot be larger than nMAX!");
69  puts("Program running is stopped !!! ");
70  exit(1);
71  }
72 
73  if(E<0.)
74  {
75  puts("Message from function RadInt2BoundFreeComplex");
76  puts("The outgoing electron energy E has to be positive!");
77  puts("Program running is stopped !!! ");
78  exit(1);
79  }
80 
81  if(sign!=-1 && sign!=+1)
82  {
83  puts("Message from function RadInt2BoundFreeBurgess:");
84  puts("sign has to be either -1 or +1!");
85  puts("Program running is stopped !!! ");
86  exit(1);
87  }
88 
89  double k=sqrt(2.*fabs(E)); // free electron momentum (velocity) in atomic units
90 
91  if(k<1.e-10) // k=0 would cause division by zero error
92  {
93  k=1.e-10;
94  }
95  double k2=k*k;
96  double D=1.+n*n*k2;
97 
98  // Eq. 30 of Burgess 1965:
99  double logg0=0.5*(logpi-LogN[2]-LogNFactor[2*n-1])+LogN[4]+n*(LogN[4]+LogN[n])-2*n;
100  // Eq. 31 of Burgess 1965:
101  double logPs=0.;
102  for(int s=1;s<=n;s++)
103  {
104  logPs+=log(1.+s*s*k2);
105  }
106  double logg1=0.5*(logPs-log(1.-exp(-2.*M_PI/k)))+2*n-2./k*atan(n*k)-(n+2)*log(D)+logg0;
107 
108  double g1=1.; // we use here g1=1 instead of exp(logg1), to avoid overflow or underflow
109  // Eq. 32 of Burgess 1965:
110  double g2=0.5*sqrt((2.*n-1.)*D)*g1;
111  // Eq. 33 of Burgess 1965:
112  double g3=1./(2*n)*sqrt(D/(1.+(n-1.)*(n-1.)*k2))*g1;
113  // Eq. 34 of Burgess 1965:
114  double g4=(4.+(n-1.)*D)/(2.*n)*sqrt((2.*n-1.)/(1.+(n-2.)*(n-2.)*k2))*g3;
115 
116 // printf("logg0,logg1= %12.3f %12.3f \t\n",logg0,logg1);
117 
118  double Exponent=logg1;
119 
120  double g = 0., RadInt;
121 
122  // We have to multiply the output by 2/M_PI, due to the different wave func. normalizations;
123  // Burgess uses a normalization of pi*delta(k^2-k'^2), and I use delta(E-E');
124  // therefore my free wave function is sqrt(2./M_PI) times larger than the free wave func. of Burgess.
125 
126  double Cfac=2./M_PI;
127 
128  if(l==n-1 && sign==1)
129  {
130  g = g1*exp(Exponent);
131  RadInt=n*n*g;
132  return RadInt*RadInt*Cfac;
133  }
134  else if(l==n-2 && sign==1)
135  {
136  g = g2*exp(Exponent);
137  RadInt=n*n*g;
138  return RadInt*RadInt*Cfac;
139  }
140  else if(l==n-1 && sign==-1)
141  {
142  g = g3*exp(Exponent);
143  RadInt=n*n*g;
144  return RadInt*RadInt*Cfac;
145  }
146  else if(l==n-2 && sign==-1)
147  {
148  g = g4*exp(Exponent);
149  RadInt=n*n*g;
150  return RadInt*RadInt*Cfac;
151  }
152 
153  double A,B,C;
154  int j;
155 
156 
157  if(sign==1)
158  {
159  for(int L=n-3;L>=l;L--)
160  {
161  j=L+2;
162  A=2.*n*sqrt((n*n-(j-1)*(j-1))*(1.+j*j*k2));
163  B=4.*n*n-4.*j*j+j*(2*j-1)*D;
164  C=-2.*n*sqrt((n*n-j*j)*(1.+(j+1)*(j+1)*k2));
165  g=(B*g2+C*g1)/A;
166  g1=g2;
167  g2=g;
168  if(fabs(g)>Chigh)
169  {
170  g*= Clow;
171  g1*= Clow;
172  g2*= Clow;
173  Exponent += Ehigh;
174  }
175  else if(fabs(g)<Clow)
176  {
177  g*= Chigh;
178  g1*= Chigh;
179  g2*= Chigh;
180  Exponent -= Ehigh;
181  }
182 // printf("L,g,Exponent= %12i %12.2e %12.3f \t\n",L,g,Exponent);
183  }
184  }
185  else
186  {
187  for(int L=n-3;L>=l;L--)
188  {
189  j=L+1;
190  A=2.*n*sqrt((n*n-j*j)*(1.+(j-1)*(j-1)*k2));
191  B=4.*n*n-4.*j*j+j*(2*j+1)*D;
192  C=-2.*n*sqrt((n*n-(j+1)*(j+1))*(1.+j*j*k2));
193  g=(B*g4+C*g3)/A;
194  g3=g4;
195  g4=g;
196  if(fabs(g)>Chigh)
197  {
198  g*= Clow;
199  g3*= Clow;
200  g4*= Clow;
201  Exponent += Ehigh;
202  }
203  else if(fabs(g)<Clow)
204  {
205  g*= Chigh;
206  g3*= Chigh;
207  g4*= Chigh;
208  Exponent -= Ehigh;
209  }
210 // printf("L,g,Exponent= %12i %12.2e %12.3f \t\n",L,g,Exponent);
211  }
212  }
213 
214  g*=exp(Exponent);
215  RadInt=n*n*g;
216 
217  // We have to multiply the output by 2/M_PI, due to the different wave func. normalizations;
218  // Burgess uses a normalization of pi*delta(k^2-k'^2), and I use delta(E-E');
219  // therefore my free wave function is sqrt(2./M_PI) times larger than the free wave func. of Burgess.
220 
221  return RadInt*RadInt*Cfac;
222  }

226  double FBBRionization::SigmaPhotoionization(int n, int l, double omega) {
227  // Ionization energy of the (n,l) state in atomic units:
228  double Enl=1./(2.*n*n);
229 
230  // Zero cross section below the ionization limit:
231  if(omega<Enl)
232  return 0.;
233 
234  // Outgoing electron energy in atomic units:
235  double E=omega-Enl;
236 
237  double Sum=0.;
238  for(int sign=-1 ;sign<=+1; sign+=2)
239  {
240  int lp=l+sign; // angular momentum quantum number of final state
241  if(lp<0) continue;
242 
243  double lplmax;
244  if(sign<0)
245  lplmax=l;
246  else
247  lplmax=l+1;
248 
249  double M2 = RadInt2BoundFreeBurgess(n, l, E, sign);
250  Sum+=lplmax*M2;
251  }
252 
253  return fC*omega/(2.*l+1.)*Sum;
254  }

258  double FBBRionization::operator()(double E) {
259  // E is the energy of the outgoing electron in atomic units
260 
261  // Ionization energy of the (n,l) state in atomic units:
262  double Enl=1./(2.*(double)(fn*fn));
263  // Photon energy in atomic units:
264  double omega=Enl+E;
265 
266  double omegakT=omega/(fAtomic_kB *fT);
267  if(omegakT>550.)
268  omegakT=550.; // to avoid overflow
269  double nbar=1./(exp(omegakT)-1.);
270 
271  // Photon number density:
272  double Nph=omega*omega/(M_PI*M_PI* fAtomic_C * fAtomic_C * fAtomic_C)*nbar;
273  // Photoionization cross section:
274  double sigma=SigmaPhotoionization(fn,fl,omega);
275 
276  return fAtomic_C *sigma*Nph;
277  }

314  if (b > 0 || c <= 0) {
315  intmsg(eError) << "Message from function HypergeometricFHoangBinh:" << ret
316  << "b>0 or c<=0!" << ret
317  << "b,c= " << b << ", " << c << ret
318  << "Program running is stopped !!! " << eom;
319  }
320 
321  double A = 1.;
322  double B = 1. - a / (double) c * x;
323 
324  if (b == 0) {
325  E = 0;
326  return A;
327  }
328 
329  if (b == -1) {
330  E = 0;
331  return B;
332  }
333 
334  E = 0;
335  double F = 0., D;
336 
337  for (int n = -2; n >= b; n--) {
338  D = 1. / (double) (n + 1 - c);

340  A = B;

342

344  F *= Clow;
345  A *= Clow;
346  B *= Clow;

362  double RydbergCalculator::RadInt2Gordon(int n, int l, int np, int sign) {
363 
364  if (n < 1 || np < 1 || l < 0 || l > n - 1) {
365  intmsg(eError)
366  << "Message from function RadInt2Gordon:" << ret
367  << "n and np have to be larger than 0!" << ret
368  << "l has to be larger than -1 and smaller than n !" << ret
369  << "Program running is stopped !!! " << eom;
370  }
371 
372  if (n > nMAX || np > nMAX) {
373  intmsg(eError)
374  << "Message from function RadInt2Gordon" << ret
375  << "n and np cannot be larger than nMAX!" << ret
376  << "Program running is stopped !!! " << eom;
377  }
378 
379  if (sign != -1 && sign != +1) {
380  intmsg(eError)
381  << "Message from function RadInt2Gordon:" << ret
382  << "sign has to be either -1 or +1!" << ret
383  << "Program running is stopped !!! " << eom;
384  }
385 
386  int lp = l + sign;
387  if (lp < 0 || lp > np - 1)
388  return 0.;
389 
390  if (sign == 1) {
391  int N, NP, L;
392  N = np;
393  NP = n;
394  L = l + 1;
395  n = N;
396  np = NP;
397  l = L;
398  }
399 
400  if (n == np) // exception!
401  {
402  return 9. / 4. * n * n * (n * n - l * l);
403  }
404 
405  int nr = n - l - 1;
406  int npr = np - l;
407 
408  // A, B, C, D, E0:
409  double A = -LogN[4] - LogNFactor[2 * l - 1];
410  double B = 0.5 * (LogNFactor[n + l] + LogNFactor[np + l - 1] - LogNFactor[n - l - 1] - LogNFactor[np - l]);
411  double C = (l + 1) * (LogN[4] + LogN[n] + LogN[np]);
412  double D = (n + np - 2 * l - 2) * LogN[abs(n - np)] - (n + np) * LogN[n + np];
413  double E0 = A + B + C + D;
414 
415  // x, del2, sum2:
416  double del2 = (n - np) * (n - np);
417  double sum2 = (n + np) * (n + np);
418  double x = -4. * n * np / del2;
419 
420  // F1, F2 :
421  double E1, E2, F1, F2;
422 
423  if (n > np) {
424  F1 = HypergeometricFHoangBinh(-nr, -npr, 2 * l, x, E1);
425  F2 = HypergeometricFHoangBinh(-nr - 2, -npr, 2 * l, x, E2);
426  }
427  else {
428  F1 = HypergeometricFHoangBinh(-npr, -nr, 2 * l, x, E1);
429  F2 = HypergeometricFHoangBinh(-npr, -nr - 2, 2 * l, x, E2);
430  }
431 
432  // RGordon:
433  double RGordon = exp(E0 + E1) * F1 - (del2 / sum2) * exp(E0 + E2) * F2;
434 
435 
436  return RGordon * RGordon;
437  }

534  void RydbergCalculator::SpontaneousEmissionGenerator(int n, int l, double &Psptotal, int &np, int &lp) {
535  // Calculation of the discrete probability distributions function values PDF[i], i=1,...,2*N.
536  // PDF[i]= Psp(n,l,np,sign) / Psptotal;
537  // Psp(n,l,np,sign): single spontaneous emission rate from initial state (n,l)
538  // to final state (np,l+sign).
539  // Psptotal: sum of all single spontaneous emission rate values.
540  // sign=-1: np goes from npmin to npmax, i from 1 to N;
541  // sign=+1: np goes from npmin to npmax, i from N+1 to 2*N;
542 
543  if (n == 1) // no decay from ground state!
544  {
545  Psptotal = 0.;
546  np = 1;
547  lp = 0;
548 
549  return;
550  }
551 
552  if (n == 2 && l == 0) // no 2s --> 1s decay !
553  {
554  Psptotal = 0.;
555  np = 2;
556  lp = 0;
557 
558  return;
559  }
560 
561  int npmin = l;
562  if (l == 0)
563  npmin = 1;
564 
565  int npmax = n - 1;
566  int N = npmax - npmin + 1; // maximal value of the discrete random variable i
567 
568  //Code from here is identical to BBRTransitionGenerator
569 
570  double PDF[2 * nMAX + 2], CDF[2 * nMAX + 2];
571 
572  int sign;
573  Psptotal = 0.;
574 
575  for (int i = 1; i <= 2 * N; i++) {
576  if (i <= N) {
577  sign = -1;
578  np = npmin + i - 1;
579  }
580  else {
581  sign = 1;
582  np = npmin + i - N - 1;
583  }
584 
585  PDF[i] = Psp(n, l, np, sign);
586  Psptotal += PDF[i];
587 
588  }
589 
590  double tau = 1. / Psptotal; // spontaneous emission lifetime
591 
592  // Normalized PDF values:
593  for (int i = 1; i <= 2 * N; i++) {
594  PDF[i] *= tau;
595  }
596 
597  // Calculation of the discrete cumulative distribution function CDF[i]:
598  CDF[1] = 0.;
599  for (int i = 2; i <= 2 * N + 1; i++) {
600  CDF[i] = CDF[i - 1] + PDF[i - 1];
601  }
602  // CDF[2*N+1] has to be = 1 !
603 
604  // random number in [0,1]:
605  double u = KRandom::GetInstance().Uniform(0., 1., true, false);
606 
607  // We search now the index value i so that CDF[i] < u < CDF[i+1],
608  // using the binary search algorithm:
609  int i = 1;
610  int j = 2 * N + 1;
611  int k;
612  do {
613  k = (i + j) / 2;
614  if (CDF[k] < u)
615  i = k;
616  else
617  j = k;
618  }
619  while (j - i > 1);
620  // Result of binary search: i
621 
622  // Control:
623  if (!(CDF[i] <= u && u <= CDF[i + 1]))
624  printf("u,i,CDF[i],CDF[i+1]= %12.5f %9i %12.5f %12.5f \t\n", u, i, CDF[i], CDF[i + 1]);
625 
626  // np and lp :
627  if (i <= N) {
628  sign = -1;
629  np = npmin + i - 1;
630  }
631  else {
632  sign = 1;
633  np = npmin + i - N - 1;
634  }
635 
636  lp = l + sign;
637 
638  return;
639 
640  }

644  void RydbergCalculator::BBRTransitionGenerator(double T, int n, int l, double &PBBRtotal, int &np, int &lp) {
645  // Calculation of the discrete probability distributions function values PDF[i], i=1,...,2*N.
646  // PDF[i]= PBBR(n,l,np,sign) / PBBRtotal;
647  // PBBR(T,n,l,np,sign): single BBR induced transition rate from initial state (n,l)
648  // to final state (np,l+sign).
649  // PBBRtotal: sum of all single BBR induced transition rate values.
650  // sign=-1: np goes from npmin to npmax, i from 1 to N;
651  // sign=+1: np goes from npmin to npmax, i from N+1 to 2*N;
652 
653 
654  int npmax = 10 * n;
655  if (n > 80)
656  npmax = 5 * n;
657 
658  if (npmax > nMAX)
659  npmax = nMAX;
660 
661  //Code from here is identical to SpontaneousEmissionGenerator
662 
663  double PDF[2 * nMAX + 2], CDF[2 * nMAX + 2];
664 
665  int npmin = l;
666  if (l == 0)
667  npmin = 1;
668 
669  int N = npmax - npmin + 1; // maximal value of the discrete random variable i
670 
671  int sign;
672  PBBRtotal = 0.;
673 
674  for (int i = 1; i <= 2 * N; i++) {
675  if (i <= N) {
676  sign = -1;
677  np = npmin + i - 1;
678  }
679  else {
680  sign = 1;
681  np = npmin + i - N - 1;
682  }
683 
684  if (np != n)
685  PDF[i] = PBBR(T, n, l, np, sign);
686  else
687  PDF[i] = 0.;
688 
689  PBBRtotal += PDF[i];
690 
691  }
692 
693  double tau = 1. / PBBRtotal; // BBR induced transition lifetime
694 
695  // Normalized PDF values:
696  for (int i = 1; i <= 2 * N; i++) {
697  PDF[i] *= tau;
